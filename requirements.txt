
# ============================================================
#   Smart Shopping Trolley with Billing System
#   Requirements / Dependencies
# ============================================================

# ------------------------------------------------------------
# SOFTWARE REQUIREMENTS
# ------------------------------------------------------------

# Arduino IDE (Programming & Uploading Firmware)
# Download: https://www.arduino.cc/en/software
Arduino IDE >= 1.8.0

# USB-to-Serial Driver (for Arduino UNO board recognition)
# Windows: CH340 or FTDI driver (depending on UNO clone)
# Download: https://www.wch-ic.com/downloads/CH341SER_EXE.html
CH340 Driver (for Arduino UNO clones) OR
FTDI Driver (for genuine Arduino UNO)


# ------------------------------------------------------------
# ARDUINO LIBRARIES (Install via Arduino Library Manager
#                    or include the folders from this project)
# ------------------------------------------------------------

# RFID Reader Library (MFRC522)
# Library Name : MFRC522
# Author       : GithubCommunity / miguelbalboa
# Version      : >= 1.4.0
# Source       : https://github.com/miguelbalboa/rfid
# Install via  : Arduino IDE → Library Manager → Search "MFRC522"
MFRC522 >= 1.4.0

# RFID Reader Library v2 (Extended / Alternative)
# Library Name : RFID_MFRC522v2
# Author       : OSSLibraries
# Version      : >= 2.0.0
# Source       : https://github.com/OSSLibraries/Arduino_MFRC522v2
RFID_MFRC522v2 >= 2.0.0

# I2C LCD Display Library
# Library Name : LiquidCrystal I2C
# Author       : Frank de Brabander
# Version      : >= 1.1.2
# Source       : https://github.com/johnrickman/LiquidCrystal_I2C
# Install via  : Arduino IDE → Library Manager → Search "LiquidCrystal I2C"
LiquidCrystal_I2C >= 1.1.2

# SPI Communication Library (Built-in with Arduino IDE)
# No separate install required
SPI (Built-in)

# I2C / Wire Communication Library (Built-in with Arduino IDE)
# No separate install required
Wire (Built-in)


# ------------------------------------------------------------
# HARDWARE REQUIREMENTS
# ------------------------------------------------------------

# Microcontroller Board
Arduino UNO (ATmega328P) x1

# RFID Reader Module
MFRC522 RFID Reader Module (13.56 MHz) x1

# RFID Cards / Tags (Passive, ISO 14443 Type A)
# Minimum 4 Cards:
#   - 3 Cards for Products (e.g., Milk, Bread, Juice)
#   - 1 Card for Checkout
RFID Card or Tag (13.56 MHz, ISO 14443A) x4 (minimum)

# LCD Display
16x2 Character LCD with I2C Backpack Module (Address: 0x27) x1

# Connecting Wires
Jumper Wires (Male-to-Male, Male-to-Female) — assorted

# Prototyping Board
Breadboard (830 points) x1

# Power Supply
USB Cable (Type-A to Type-B) x1     — for Arduino power via PC
  OR
5V DC Power Adapter / Battery Pack   — for standalone operation


# ------------------------------------------------------------
# PIN CONNECTIONS SUMMARY
# ------------------------------------------------------------

# MFRC522 → Arduino UNO
#   SDA (SS)  →  Pin 10
#   SCK       →  Pin 13
#   MOSI      →  Pin 11
#   MISO      →  Pin 12
#   RST       →  Pin 9
#   GND       →  GND
#   3.3V      →  3.3V  ⚠️ WARNING: Do NOT connect to 5V

# 16x2 I2C LCD → Arduino UNO
#   VCC  →  5V
#   GND  →  GND
#   SDA  →  A4
#   SCL  →  A5


# ------------------------------------------------------------
# OPTIONAL / FUTURE UPGRADE REQUIREMENTS
# ------------------------------------------------------------

# For Wi-Fi / Cloud Integration
ESP8266 NodeMCU or ESP32 Module x1

# For Mobile App Development
Flutter SDK >= 3.0.0   OR   Android Studio

# For Weight-Based Anti-Theft
HX711 Load Cell Amplifier x1
Load Cell Sensor (1kg or 5kg) x1

# For OLED Display Upgrade
SSD1306 OLED Display (128x64, I2C) x1
# Library: Adafruit SSD1306 >= 2.5.0
Adafruit SSD1306 >= 2.5.0
Adafruit GFX Library >= 1.11.0
